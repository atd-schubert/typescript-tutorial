Aufbau - Unterschiede

JavaScript (ES5)

                            var irgendwas = 'Ein Wert';

function tueEtwas(parameter) {
    console.log(parameter);
}

var anonymeFunktion = function () {
    // eine anonyme Funktion
};

MeineKlasse.prototype = {
    '__proto__': EventEmitter,
    methode: function (bool) {
        // eine Methode...
        return bool;
    }
};

function MeineKlasse(paramerter) {
    this.parameter = parameter;
    // Das ist der Constructor
}

JavaScript (ES6)

                            let irgendwas = 'Ein Wert';

function tueEtwas(parameter) {
    console.log(parameter);
}
let anonymeFunktion = () => {
    // eine anonyme Funktion
};

class MeineKlasse extends EventEmitter {
    constructor (parameter) {
        this.parameter = parameter;
        // Das ist hier der Constructor
    }

    methode (bool) {
        // eine Methode...
        return bool;
    }
}

TypeScript

                            let irgendwas: string = 'Ein Wert';

function tueEtwas(parameter: any): void {
    console.log(parameter);
}
let anonymeFunktion: (parameter: any) => void = (parameter: any): void => {
   // eine anonyme Funktion
};

interface IMeineKlasse {
    parameter: string;
}

class MeineKlasse extends EventEmitter implements IMeineKlasse {
    constructor (public parameter: any) {
        // Das ist hier der Constructor
    }

    public methode (bool: boolean): boolean {
        // eine Methode...
        return bool;
    }
}

Interfaces

Einfaches Interface

interface ITileLayerOptions {
    url: string;
    attribution: string;
    maxZoomLevel: number;
    minZoomLevel: number
}
var opts: ITileLayerOptions = {
    attribution: 'Ein fiktiver Tile-Server',
    url: 'http://tile.bla.de/{z}/{x}/{y}.png'
};

Interface mit optionalen Eigenschaften

interface ITileLayerOptions {
    url: string;
    attribution?: string;
    maxZoomLevel?: number;
    minZoomLevel?: number
}
var opts: ITileLayerOptions = {
    attribution: 'Ein fiktiver Tile-Server',
    url: 'http://tile.bla.de/{z}/{x}/{y}.png'
};

Interface mit Constructor oder für eine Funktion

interface IMyMathFunction {
    (a: number, b: number): number;
};
var addition: IMyMathFunction = function (a: number, b: number): number {
    return a + b;
}
var multiplication: IMyMathFunction = (a: number, b: number): number => {
    return a * b;
}
Inline (ohne Interface)

var fn: (x: number, y: number)=> number = (x, y) => { return x + y; }

Klassen

Public

öffentlich erreichbare Eigenschaften:


class GrosseKlasse {
    public eigenschaft: any;

    public methode (...zahlen: number[]): string {
        let ergebnis: number = 0;
        for (let i = 0; i < zahlen.length; i += 1) {
            ergebnis += zahlen[i];
        }
        return this.offentlich + ergebnis.toString();
    }

    constructor (public offentlich: string) {}
}

Private

Nur innerhalb der Klasse erreichbare Eigenschaften


class GrosseKlasse {
    private eigenschaft: any;

    private getNichtOffentlich (): string {
        return this.nichtOffentlich
    }
    private setNichtOffentlich (value: string): void {
        this.nichtOffentlich = value
    }

    constructor (private nichtOffentlich: string) {}
}

Static

Eigenschaften der Klasse, nicht des Objekts


class GrosseKlasse {
    public static factory (): GrosseKlasse {
        return new GrosseKlasse();
    }
}

Vererbung


class KleineKlasse extends GrosseKlasse {
    public spezial (): void {
        // Tue etwas
    }
}

Mehrfach Vererbung


class KleineKlasse extends EventEmitter, GrosseKlasse {
    public spezial (): void {
        // Tue etwas
    }
}

Mehrfach implementiert und vererbt


class KleineKlasse extends EventEmitter, GrosseKlasse implements IKleineKlasse, ISpezial {
    public spezial (): void {
        // Tue etwas
    }
}
